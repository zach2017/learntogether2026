version: '3.8'

services:
  keycloak:
    image: quay.io/keycloak/keycloak:23.0
    container_name: keycloak
    environment:
      KEYCLOAK_ADMIN: admin
      KEYCLOAK_ADMIN_PASSWORD: admin
      KC_HTTP_PORT: 8080
    ports:
      - "8080:8080"
    command:
      - start-dev
    networks:
      - spring-keycloak-network
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8080/health/ready"]
      interval: 30s
      timeout: 10s
      retries: 5

  resource-server:
    build:
      context: ./resource-server
      dockerfile: Dockerfile
    container_name: resource-server
    ports:
      - "8081:8081"
    environment:
      SPRING_PROFILES_ACTIVE: docker
      KEYCLOAK_URL: http://keycloak:8080
    depends_on:
      keycloak:
        condition: service_healthy
    networks:
      - spring-keycloak-network

  client-app:
    build:
      context: ./client-app
      dockerfile: Dockerfile
    container_name: client-app
    ports:
      - "8082:8082"
    environment:
      SPRING_PROFILES_ACTIVE: docker
      KEYCLOAK_URL: http://keycloak:8080
      RESOURCE_SERVER_URL: http://resource-server:8081
    depends_on:
      - keycloak
      - resource-server
    networks:
      - spring-keycloak-network

networks:
  spring-keycloak-network:
    driver: bridge
